TITLE Addressing Modes		(part2.asm)

include irvine32.inc

.stack 4096

.data
	PromptUser BYTE "Please enter a value: ", 0
	DisplayText BYTE "Value: ", 0
	CarriageReturn BYTE " ", 13, 10, 0

.code
main proc
	mov ecx, 5
	mov eax, 0
	L1:
		mov edx, offset PromptUser
		call writestring
		call readint
		push eax
		loop L1

	mov ecx, 5
	L2:
		mov edx, offset DisplayText
		call writestring
		pop eax
		call writeint
		mov edx, offset CarriageReturn
		call writestring
		loop L2
	
	invoke ExitProcess, 0
	main endp

end main
